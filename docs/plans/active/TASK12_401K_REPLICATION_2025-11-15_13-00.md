```python
"""
Execute 401(k) replication and generate comparison report.

Runs the Chernozhukov et al. (2018) 401(k) replication and compares
our results with published estimates.
"""

import argparse
from datetime import datetime
import pandas as pd
from pathlib import Path

from src.validation.empirical_replication import (
    FourZeroOneKReplication,
    load_401k_data,
)


def main(
    data_path: Optional[str] = None,
    treatment: str = "e401",
    output_dir: str = "results/empirical",
    methods: List[str] = ["PLR_RandomForest"],
):
    """Run 401(k) replication and save results."""

    # Load data
    if data_path:
        df = pd.read_csv(data_path)
    else:
        df = load_401k_data(source="econml")

    # Initialize replication
    rep = FourZeroOneKReplication(
        data_path=data_path or "from_econml",
        treatment=treatment,
        n_folds=3,
        random_state=42,
    )

    # Run replications
    results = []
    if "PLR_RandomForest" in methods:
        result = rep.replicate_plr_rf()
        results.append(result)

    if "PLR_Lasso" in methods:
        result = rep.replicate_plr_lasso()
        results.append(result)

    # Create comparison DataFrame
    comparison = pd.DataFrame([
        {
            "method": r.method,
            "our_ate": r.ate_estimate,
            "our_se": r.std_error,
            "our_ci_lower": r.ci_lower,
            "our_ci_upper": r.ci_upper,
            "published_ate": r.published_ate,
            "difference": r.difference,
            "rel_difference_pct": r.rel_difference * 100,
            "p_value": r.p_value,
            "status": r.status,
        }
        for r in results
    ])

    # Save results
    Path(output_dir).mkdir(parents=True, exist_ok=True)
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    output_path = f"{output_dir}/401k_replication_{timestamp}.csv"
    comparison.to_csv(output_path, index=False)

    # Print summary
    print("\n=== 401(k) Replication Results ===\n")
    print(comparison.to_string(index=False))
    print(f"\nResults saved to: {output_path}")

    return comparison


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Run 401(k) replication")
    parser.add_argument("--data", type=str, help="Path to 401(k) dataset CSV")
    parser.add_argument("--treatment", type=str, default="e401", help="Treatment variable")
    parser.add_argument("--output", type=str, default="results/empirical", help="Output directory")
    args = parser.parse_args()

    main(data_path=args.data, treatment=args.treatment, output_dir=args.output)
```
